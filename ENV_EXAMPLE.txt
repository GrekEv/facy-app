# Токен бота от BotFather
BOT_TOKEN=1234567890:ABCdefGHIjklMNOpqrsTUVwxyz

# URL вашего Web App (ОПЦИОНАЛЬНО - можно использовать домен или Railway/Render/ngrok URL)
# Варианты:
# 1. С доменом: WEBAPP_URL=https://your-domain.com
# 2. Railway: WEBAPP_URL=https://your-app.up.railway.app (или оставьте пустым - определится автоматически)
# 3. Render: WEBAPP_URL=https://your-app.onrender.com (или оставьте пустым - определится автоматически)
# 4. ngrok: WEBAPP_URL=https://xxx.ngrok-free.app
# 5. Оставить пустым - будет использован автоматически определенный URL или http://localhost:8000

# ============================================
# API КЛЮЧИ ДЛЯ ЗАМЕНЫ ЛИЦ
# ============================================

# DeepFace Video (рекомендуется)

DEEPSWAP_API_URL=https://api.deepswap.ai

# ============================================
# API КЛЮЧИ ДЛЯ ГЕНЕРАЦИИ ИЗОБРАЖЕНИЙ
# ============================================

# Replicate (рекомендуется для изображений)
# Зарегистрируйтесь на https://replicate.com
# Account → API Tokens → Create token
REPLICATE_API_KEY=your_replicate_api_key
REPLICATE_API_URL=https://api.replicate.com/v1

# ============================================
# API КЛЮЧИ ДЛЯ ГЕНЕРАЦИИ ВИДЕО
# ============================================

# OpenAI Sora (для генерации видео высокого качества)
# Зарегистрируйтесь на https://platform.openai.com
# API Keys → Create new secret key
OPENAI_API_KEY=your_openai_api_key
SORA_MODEL=sora-1.0-pro  # или sora-2.0

# Higgsfield.ai (для генерации видео)
# Зарегистрируйтесь на https://higgsfield.ai
# API → Create new API key
HIGGSFIELD_API_KEY=your_higgsfield_api_key
HIGGSFIELD_API_URL=https://api.higgsfield.ai

# Replicate (также поддерживает генерацию видео)
# Используйте тот же REPLICATE_API_KEY что и для изображений

# ============================================
# ВЫБОР ПРОВАЙДЕРОВ
# ============================================

# Для генерации изображений
IMAGE_GENERATION_PROVIDER=replicate  # replicate

# Для генерации видео (выберите один)
VIDEO_GENERATION_PROVIDER=replicate  # replicate, sora, higgsfield

# Примечание: Если API ключи не указаны, используется mock режим для тестирования

# База данных
# Оставьте пустым для SQLite (по умолчанию)
# Или укажите PostgreSQL URL от Railway (если создали базу данных)
DATABASE_URL=

# Режим работы (production для продакшена)
ENVIRONMENT=production

# ID администраторов через запятую
ADMIN_IDS=123456789,987654321
